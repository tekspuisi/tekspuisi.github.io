Kata baku biasanya digunakan pada situasi yang resmi baik berupa tulisan seperti surat, laporan, buku dan sebagainya, maupun berupa pengungkapan atau perkataan seperti tes wawancara, diskusi secara formal dan sebagainya.

38. Tutorial membuat game tebak Kata dengan bahasa c++


Berikut programnya, untuk mengubah menjadi 1-100 kamu hanya perlu mengubah "rand()%100"
#include <iostream>
#include <cstdlib>
using namespace std;
int main()
{
    int arr[1];
    arr[0]=rand()%10;
    cout << "-----------Tebak Angka dari 1-10 (Cari angka X)--------\n";

    for (int i=2; i>=0; i--)
    {

        cout << "Kesempatan mu untuk menebak adalah sebanyak " << i+1 << "kali\n\nMasukkan Angka: ";
        int tebak;
        cin >> tebak;
        if (arr[i]==tebak)
        {
            cout << "Angka X adalah " << tebak << "\nTebakanmu benar!\n";
            return 0;
        }
        else if (i>0) cout << "\nAngka yang ditebak salah!\n" << endl;
        else cout << "\nKamu gagal menebak angka.\n";
    }
}
